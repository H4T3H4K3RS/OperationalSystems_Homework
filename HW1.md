# **Задача:** Сгенерировать случайное число от 0 до 100 и вывести результат  

## Решение

```bash
#!/bin/bash

# Сгенерировать случайное число от 0 до 100
# и вывести результат

# Определите функцию для генерации случайного числа
function random_gen {
  # Генерируем случайное число от 0 до 1
  random_num=$(echo "scale=2; $RANDOM/32767" | bc -l)
  # Умножьте на 100, чтобы получить число от 0 до 100
  result=$(echo "scale=2; $random_num*100" | bc -l)
  # Выведите результат
  echo "$result"
}

# Цикл While для проверки результата
while [ true ]
do
  # Вызов функции
  result=$(random_gen)
  # Проверьте, находится ли результат между 0 и 100
  if [[ $(echo "scale=2; $result > 0" | bc -l) -eq 1 && $(echo "scale=2; $result < 100" | bc -l) -eq 1 ]]
  then
    # Выведите результат, если он находится между 0 и 100
    echo "Случайное число - $result"
    break
  fi
done

```
